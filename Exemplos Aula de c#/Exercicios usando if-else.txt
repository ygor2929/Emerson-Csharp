3.Calculadora Simples (If-Else): Crie um programa que receba dois números e uma

operação (+, -, *, /) como entrada (armazenados em variáveis). Use uma série de if-
else if para realizar a operação selecionada e exibir o resultado. Lembre-se de tratar

a divisão por zero.

R:
using System;
using System.Collections.Generic;
using System.Linq;
using System.Text;
using System.Threading.Tasks;
using static System.Console;

namespace exercicio3
{
    internal class Program
    {
        static void Main(string[] args)
        {
         double num1, num2, result;
            string op;

            Write("Digite o primeiro numero:");
            num1 = double.Parse(ReadLine());

            Write("Digite o primeiro numero:");
            num2 = double.Parse(ReadLine());

            Write("Digite a operação (+, -, *, /): ");
            op = ReadLine();

            if (op == "+")
            {
                result = num1 + num2;
                Write($"O resultado de {num1} + {num2} é:{result}");
            }
            else if (op == "-")
            {
                result = num1 - num2;
                Write($"O resultado de {num1} - {num2} é:{result}");
            }
            else if (op == "*")
            {
                result = num1 * num2;
                Write($"O resultado de {num1} * {num2} é:{result}");
            }
            else if (op == "/")
            {
                result = num1 + num2;
                if (num2 != 0)
                {
                    Write($"O resultado de {num1} / {num2} é:{result}");
                }
                else
                {
                    Write("Erro: Divisão por zero não é permitida!");
                }
                
            }
            else
            {
                Write("Operação invalida");
            }
            ReadKey();
        }
    }
}

            {
                result = num1 * num2;

            else
            {
                if (num1 > num2)
                
                    result = num1 - num2;
